@startuml
skinparam monochrome true

actor Сотрудник as u
participant "Фронтенд" as w
participant "Бэкенд" as b
database "База данных" as d

u -> w : Ввод данных для создания документа 
activate w
w -> w : Обработка введенной информации
w -> b : Обработанные данные документа
activate b
b -> d : Запись данных
activate d
d -> b : Бинарный код об успешной записи
deactivate d
b -> b : Форматирование сообщения UTF-8
b -> w : Текстовая строка
deactivate b
w -> u : Сообщение об успешном добавлении 
deactivate w

u -> w : Вход в систему
activate w
w -> b : Данные пользователя
activate b
b -> d : Запрос информации пользователя по логину 
activate d
d -> b : Логин и пароль 
deactivate d
b -> b : Проверка на существование логина и совпадение пароля 
b -> d : Запрос метаданных документов
activate d
d -> b : Метаданные документов
deactivate d
b -> w : Метаданные документов
deactivate b
w -> w : Обработка и группировка данных
w -> u : Отображение KANBAN-доски
deactivate w

u -> w : Выбор интересующего пользователя документа
activate w
w -> b : ID выбранного документа и уровень доступа пользователя
activate b
b -> d : Запрос информации документа по его ID 
activate d
d -> b : Информации по требуемому документу
deactivate d
b -> b : Обработка запроса в соотвествии с уровнем доступа пользователя
b -> b : Повторная обработка с учетом стадии документа
b -> w : Обработанная информация по документу
deactivate b
w -> u : Вывод атрибутов и содержания документа
deactivate w
@enduml
